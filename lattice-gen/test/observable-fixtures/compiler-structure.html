The monadic axes below are written using > on purpose. In lattice-gen, that changes the generated representation: the base descriptor code first reverses the logical lattice and then computes the join-irreducible bit encoding and closure schedules on that reversed native order. So a declaration like Uniqueness = [ Aliased > Unique ] emits aliased = 0 and unique = 1, and Contention = [ Contended > Shared > Uncontended ] emits contended = 0, shared = 1, uncontended = 3. That lets the compiler-facing monadic fragment use the intended stored polarity directly. This is separate from the DSL's semantic ^op: plain Uniqueness still denotes the same logical order as before, so the compiler bridge morphs below still spell ^op where they mean the opposite order.

  let[@inline] imply x y =

  type view = {
    uniqueness : Uniqueness.t;
    contention : Contention.t;
    visibility : Visibility.t;
    staticity : Staticity.t;
  }

  type nonrec product = t

  let of_view view =
    let uniqueness = (view.uniqueness land 1) lsl 0 in
    let contention = (view.contention land 3) lsl 1 in
    let visibility = (view.visibility land 3) lsl 3 in
    let staticity = (view.staticity land 1) lsl 5 in
    uniqueness lor contention lor visibility lor staticity

  let view t =
    {
      uniqueness = (t lsr 0) land 1;
      contention = (t lsr 1) land 3;
      visibility = (t lsr 3) land 3;
      staticity = (t lsr 5) land 1;
    }

  let make =
    fun ~uniqueness ->
    fun ~contention ->
    fun ~visibility ->
    fun ~staticity ->
    of_view {
      uniqueness;
      contention;
      visibility;
      staticity;
    }

  let proj_uniqueness t = (t lsr 0) land 1
  let with_uniqueness x t = (t land lnot 1) lor ((x land 1) lsl 0)

  let contention_bot = min_with_contention Contention.bottom

  let visibility_top = max_with_visibility Visibility.top

  let max_with_staticity x = with_staticity x top
  let staticity_bot = min_with_staticity Staticity.bottom
  let staticity_top = max_with_staticity Staticity.top

  module Axis = struct
    type _ t = ..
    type _ t += Uniqueness : Uniqueness.t t
    type _ t += Contention : Contention.t t
    type _ t += Visibility : Visibility.t t
    type _ t += Staticity : Staticity.t t
    type packed = Pack : 'a t -> packed

    let packed axis = Pack axis

    let all =
      [
        Pack Uniqueness;
        Pack Contention;
        Pack Visibility;
        Pack Staticity;
      ]

    let name : type a. a t -> string = function
      | Uniqueness -> "uniqueness"
      | Contention -> "contention"
      | Visibility -> "visibility"
      | Staticity -> "staticity"
      | _ -> invalid_arg "unknown axis"

    let pp ppf axis = Format.pp_print_string ppf (name axis)

    let proj : type a. a t -> product -> a = function
      | Uniqueness -> proj_uniqueness
      | Contention -> proj_contention
      | Visibility -> proj_visibility
      | Staticity -> proj_staticity
      | _ -> invalid_arg "unknown axis"

    let set : type a. a t -> a -> product -> product = function
      | Uniqueness -> with_uniqueness
      | Contention -> with_contention
      | Visibility -> with_visibility
      | Staticity -> with_staticity
      | _ -> invalid_arg "unknown axis"

    let min_with : type a. a t -> a -> product = function
      | Uniqueness -> min_with_uniqueness
      | Contention -> min_with_contention
