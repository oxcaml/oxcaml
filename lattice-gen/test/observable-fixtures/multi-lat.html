module Locality = struct
  type view =
    | Global
    | Local

  let global = 0
  let local = 1

  type t = int

  let min = 0
  let max = 0b1
  let[@inline] le x y = (x land y) = x
  let[@inline] equal (x : t) (y : t) = x = y
  let[@inline] join x y = x lor y
  let[@inline] meet x y = x land y
  let[@inline] subtract x y = x land lnot y
  let[@inline] imply x y = (x lxor 0b1) lor y

  module Repr = struct
    let[@inline] to_int_unsafe x = x
    let[@inline] from_int_unsafe x = x
  end

  let[@inline] view = function
    | 0 -> Global
    | 1 -> Local
    | _ -> raise_unknown_lattice_element ()

  let[@inline] of_view = function
    | Global -> global
    | Local -> local

  let print ppf x =
    let s =
      match x with
      | 0 -> "global"
      | 1 -> "local"
      | _ -> raise_unknown_lattice_element ()
    in
    Format.pp_print_string ppf s
end
