Proposed Morph Syntax

The monadic axes in this sketch use > directly. In the generator that is a representation choice, not just pretty syntax: build_base_descriptor first reverses the logical lattice and then assigns join-irreducible bits in that reversed native order. So Uniqueness = [ Aliased > Unique ] emits aliased = 0 and unique = 1, while Staticity = [ Dynamic > Static ] emits dynamic = 0 and static = 1. That makes the internal monadic representation line up with the intended polarity. This is separate from semantic ^op: plain Uniqueness still means the same logical lattice, so the bridge morphs below keep using ^op where they need the opposite order.

  let[@inline] with_staticity x t = (t land 1) lor (x lsl 1)

  let print ppf x =
    let view = view x in
    Format.fprintf
      ppf
      "%a,%a"
      Uniqueness_op.print view.uniqueness
      Staticity_op.print view.staticity
end

module Morphs = struct
  (* Primitive morphs *)
  let[@inline] locality_as_regionality x =
    x lor (x lsl 1)

  let[@inline] local_to_regional x =
    x

  let[@inline] regional_to_local x =
    x land 1

  let[@inline] regional_to_global x =
    x lsr 1

  let[@inline] regional_r2g x =
    (x lsr 1) lor (x land 2)

  let[@inline] global_to_regional x =
    1 lor (x lsl 1)

  let[@inline] linear_to_unique x =
    x

  let[@inline] unique_to_linear x =
    x


  (* Product morphs *)
  let[@inline] comonadic_locality_as_regionality x =
    (x land 1) lor (x lsl 1)

  let[@inline] comonadic_local_to_regional x =
    (x land 1) lor ((x land 6) lsl 1)

  let[@inline] comonadic_regional_to_local x =
    ((x land 12) lsr 1) lor (x land 1)

  let[@inline] comonadic_regional_to_global x =
    x lsr 1

  let[@inline] comonadic_r2g x =
    ((x land 2) lsr 1) lor (x land 14)

  let[@inline] comonadic_to_monadic_min x =
    x land 1

  let[@inline] comonadic_to_monadic_max x =
    2 lor (x land 1)

  let[@inline] monadic_to_comonadic_min x =
    x land 1

  let[@inline] monadic_to_comonadic_max x =
    2 lor (x land 1)

  let[@inline] alloc_as_value x =
    (x land 1) lor (x lsl 1)

  let[@inline] alloc_to_value_l2r x =
    (x land 1) lor ((x land 30) lsl 1)

  let[@inline] value_to_alloc_r2l x =
    ((x land 60) lsr 1) lor (x land 1)

  let[@inline] value_to_alloc_r2g x =
    x lsr 1

  let[@inline] value_r2g x =
    ((x land 2) lsr 1) lor (x land 62)

end


module type Lattice = sig
  type t
  type view

  val view : t -> view
  val of_view : view -> t

  val le : t -> t -> bool
  val equal : t -> t -> bool
  val min : t
  val max : t
  val join : t -> t -> t
  val meet : t -> t -> t
  val subtract : t -> t -> t
  val imply : t -> t -> t

  val print : Format.formatter -> t -> unit

  module Repr : sig
    val to_int_unsafe : t -> int
    val from_int_unsafe : int -> t
  end
end

(* Base Lattices *)

module Locality : sig
  (* Order:
     global < local
  *)

  type view =
    | Global
    | Local

  include Lattice with type view := view

  val global : t
  val local : t
end
